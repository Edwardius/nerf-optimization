 nerf-pytorch-10-64:
    build:
      context: .
      dockerfile: docker/nerf-pytorch/nerf-pytorch.Dockerfile
      args:
        USER_WANDB_MODE: ${WANDB_MODE:-offline}
        USER_WANDB_KEY: ${WANDB_KEY:-}

    image: vip-nerf/nerf-pytorch

    user: ${FIXUID:?}:${FIXGID:?}
    volumes:
      - ./src/nerf-pytorch:/home/docker/nerf-pytorch
    ipc: host
    tty: true
    command: ["python3", "run_nerf.py", "--config", "configs/trex_configs_def64/trex_10.txt", "--netwidth", "56", "60", "64"]
    deploy:
      resources:
        reservations:
          devices:
          - driver: nvidia
            device_ids: ['1'] # Change to ['0', '1'] to use two GPUs
            capabilities: [gpu]
